cmake_minimum_required(VERSION 3.0.0)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
project(PlayEngine)

#add_subdirectory("")
include_directories("../ExternalLibs/Assimp/include"
                    "../ExternalLibs/Bullet/include"
                    "../ExternalLibs/glew-2.1.0/include"
                    "../ExternalLibs/glfw-3.3.bin.WIN32/include"
					"../ExternalLibs/glm/glm"
					"../ExternalLibs/lua/include"
					"../ExternalLibs/SDL2_mixer-2.0.4/include"
					"../ExternalLibs/std_image/include"
                    "../ExternalLibs/SDL2-2.0.12/include"
                    "include/")

set(CMAKE_BINARY_DIR "../")
set(LIBRARY_OUTPUT_PATH ${CMAKE_BINARY_DIR})

file(GLOB_RECURSE SRC_BASE
                              "Base/*.cpp"
                              "include/Base/*.h"
)

file(GLOB_RECURSE SRC_AUDIO
                              "Audio/*.cpp"
                              "include/Audio/*.h"
)

file(GLOB_RECURSE SRC_PHYSICS
                              "Physics/*.cpp"
                              "include/Physics/*.h"
)

file(GLOB_RECURSE SRC_GRAPHICS
                              "Graphic/*.cpp"
                              "Graphic/Lights/*.cpp"
                              "include/Graphic/*.h"
                              "include/Graphic/Lights/*.h")

source_group("Audio" FILES ${SRC_AUDIO})
source_group("Base" FILES ${SRC_BASE})
source_group("Physics" FILES ${SRC_PHYSICS})
source_group("Graphics" FILES ${SRC_GRAPHICS})

add_library(PlayEngine SHARED ${SRC_BASE} ${SRC_AUDIO} ${SRC_PHYSICS} ${SRC_GRAPHICS})

target_link_libraries(PlayEngine "../../ExternalLibs/SDL2-2.0.12/lib/x86/SDL2"
                                 "../../ExternalLibs/SDL2_mixer-2.0.4/lib/x86/SDL2_mixer"
                                 "../../ExternalLibs/lua/lib/lua51"
                                 "../../ExternalLibs/glfw-3.3.bin.WIN32/lib-vc2019/glfw3"
                                 "../../ExternalLibs/glew-2.1.0/lib/Release/Win32/glew32"
                                 "../../ExternalLibs/Assimp/assimp-vc142-mtd"
                                 "Opengl32"
                                 debug "../../ExternalLibs/Bullet/lib/BulletDynamics_Debug"
                                 debug "../../ExternalLibs/Bullet/lib/Bullet3Dynamics_Debug"
                                 debug "../../ExternalLibs/Bullet/lib/BulletCollision_Debug"
                                 debug "../../ExternalLibs/Bullet/lib/LinearMath_Debug"
                                 "../../ExternalLibs/Bullet/lib/BulletDynamics"
                                 "../../ExternalLibs/Bullet/lib/Bullet3Dynamics"
                                 "../../ExternalLibs/Bullet/lib/BulletCollision"
                                 "../../ExternalLibs/Bullet/lib/LinearMath"
                                 )
